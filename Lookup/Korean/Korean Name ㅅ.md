  if (Array.isArray(v)) return v.join(", ");
  return String(v);

let links = dv.current().file.outlinks;

    seen.add(p.file.path);
    return true;
  })
  .sort((a, b) => norm(a.korean).localeCompare(norm(b.korean), "ko"));

  pages.map(p => [
